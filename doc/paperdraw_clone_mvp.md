class SimEngine {
  private intervalId: number | null = null;
  private globalRPS = 1000;

  start(graph: Graph, onTick: (state: SimState) =&gt; void) {
    this.intervalId = setInterval(() =&gt; {
      const state = this.tick(graph);
      onTick(state);
    }, 100); // 10 fps
  }

  private tick(graph: Graph): SimState {
    // 1. Distribute incoming RPS from entry points
    // 2. Propagate through graph edges (BFS/topological sort)
    // 3. Apply latency + capacity constraints per node
    // 4. Flag bottlenecks (currentRPS &gt; capacity * 0.9)
    // 5. Calculate error rates for overloaded nodes
    // 6. Return updated metric snapshot
  }
}</p><h3><strong>Bottleneck Detection Algorithm</strong></h3><ul><li>Traverse graph in topological order from client nodes</li><li>At each node: incoming_rps = sum of all upstream edge RPS</li><li>If incoming_rps &gt; node.capacity: mark as bottleneck, compute overflow</li><li>Overflow triggers error_rate increase and downstream RPS reduction</li><li>Load balancer nodes distribute RPS evenly across downstream targets</li><li>Cache nodes absorb configurable % of reads (cache hit rate)</li></ul><h2><strong>3.5 Animation Layer</strong></h2><p>Packet animation should be CSS-based (not canvas) to keep performance high:</p><ul><li>Each active edge gets a CSS class that triggers a moving dot animation</li><li>Animation speed scales linearly with edge RPS (more traffic = faster dots)</li><li>Bottlenecked edges turn red; normal edges are blue/green</li><li>React Flow's built-in animated edge prop handles the base animation</li></ul><h1><strong>4.
